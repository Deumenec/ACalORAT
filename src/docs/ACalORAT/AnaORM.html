 19def extract_kicks(dkicks, ind_old, ind_new):
 20    """
 21    Used to get the kicks inside of sexupole correctors.
 22    
 23    Takes a Jacobian of kicks in the second dimension in a given set of elements and extracts the
 24    kicks in another set of elements by matching repeated elements, suposing the rest of kicks are zero!
 25    """
 26    dkicks = np.transpose(dkicks)
 27    new_dkicks = np.zeros(np.shape(dkicks))
 28    for i, eli in enumerate(ind_old):
 29        for j , elj in enumerate(ind_new):
 30            if elj > eli:
 31                break
 32            if elj == eli:
 33                new_dkicks[j] = dkicks[i]
 34                
 35    return np.transpose(new_dkicks)

109    def dRij_dqk_thin(self, Ei : Elements, Ej : Elements, Ek : Elements):
110        """Considers all elements as thin, results can be greatlly improved by passing average
111        optics computed with the average method instead of the entrance optics, but for thick
112        elements one has to use the thick version of the formula.
113        """
114        Cij1 = self.Cabn(Ei, Ej, 1)
115        Cik2 = self.Cabn(Ei, Ek, 2)
116        Cjk2 = self.Cabn(Ej, Ek, 2)
117        Sij1 = self.Sabn(Ei, Ej, 1)
118        Sik2 = self.Sabn(Ei, Ek, 2)
119        Sjk2 = self.Sabn(Ej, Ek, 2)
120            
121        cosTerm = Cij1 * ( Cik2 + Cjk2 + 2* np.cos(np.pi * self.tune)**2)
122        sinTerm = Sij1 * ( Sik2 - Sjk2 + np.sin( 2*np.pi*self.tune)*(2*np.heaviside(Ei.muB-Ek.muB, 0)
123                    -2*np.heaviside(Ej.muB-Ek.muB, 0)-np.sign(Ei.muB-Ej.muB)))
124        
125        ana_dORM_dq = self.sgn * (
126        np.sqrt(Ei.betaB * Ej.betaB) * Ek.betaB * Ek.LengthB
127        / (8 * np.sin(np.pi * self.tune)* np.sin(2 * np.pi * self.tune)) 
128        * (cosTerm + sinTerm))
129        return np.real(ana_dORM_dq)
