148def compute_average_dispersion(ring, ind, all_disp):
149    """ Calculates the average dispersion in a set of ring elements.
150    In the horizontal direction
151    Handles thick elements, thin elements, and zero-focusing correctors/drifts safely.
152    """
153    avDispersion = np.zeros(len(ind))
154    dispersion = all_disp[ind, 0]
155    dispersionp = all_disp[ind, 1]
156    
157    for i, idx in enumerate(ind):
158        el = ring[idx]
159        length = getattr(el, "Length", 0.0)
160        bend = getattr(el, "BendingAngle", 0.0)
161        k = getattr(el, "K", 0.0)
162        
163        # If it's a thin element (length = 0), average dispersion is just point dispersion
164        if length == 0:
165            avDispersion[i] = dispersion[i]
166            continue
167        
168        k_total = k + (bend/length)**2
169        
170        # Prevent 0/0 division (NaNs) for standard correctors/drifts
171        if abs(k_total) < 1e-12:
172            # Analytical limit of integration when k -> 0
173            avDispersion[i] = dispersion[i] + dispersionp[i] * length / 2.0 + bend * length / 6.0
174        else:
175            k_c = complex(k_total)
176            phi = np.sqrt(k_c) * length
177            term1 = dispersion[i] * np.sin(phi) / phi
178            term2 = dispersionp[i] / (k_c * length) * (1 - np.cos(phi))
179            term3 = bend / (length * k_c) * (1 - np.sin(phi) / phi)
180            avDispersion[i] = np.real(term1 + term2 + term3)
181            
182    return avDispersion

187    """
188    Parameters
189    ----------
190    ring : at.lattice
191        Ring for which the matrix is calculated
192    ind_bpm : array
193        indices of the BPMs for the ORM matrix
194    ind_cor : array

197        indices

199        "v" for the vertical dimension and "h"
200    Returns
201    -------
202    num_dORM_dq: np.array 
203        The dORM_dq rank 3 tensor with indices dORM_dq[quadrupole][bpm][corrector]
204        
205    """

324def kick_cor(ring , ind_bpm, ind_cor, threshold, original_orbit):
325    """
326    Tweek corrector kickangles to minimize errors at bpms with l2 norm matching
327    the ring orbit with the original one, it does not apply any energy correction
328    """
329    max_steps = 4
330    
331    #Total kicks performed
332    t_kicks = {"h":np.zeros(len(ind_cor["h"])), "v": np.zeros(len(ind_cor["v"]))}
333    
334    #Kicks for current iteration
335    kicks = {"h":np.zeros(len(ind_cor["h"])), "v": np.zeros(len(ind_cor["v"]))}
336    
337    #Finds orbit at the beginig
338    orbit = at.find_orbit6(ring, refpts=ind_bpm)[1]
339    #Here SUPER IMPORTANT, remember to select only bpms from the original closed orbit!
340    dxs = np.array([orbit[i][0] -original_orbit[ind_bpm[i]][0] for i in range(len(orbit))])
341    dys = np.array([orbit[i][2] -original_orbit[ind_bpm[i]][2] for i in range(len(orbit))])
342    difference = rms(dxs)+rms(dys)
343    if difference<threshold:
344        #print("No correction was needed")
345        return t_kicks, orbit 
346    
347    while max_steps >1:
348        
349        max_steps -=1
350        Resp = at.latticetools.OrbitResponseMatrix(ring,"h", ind_bpm, ind_cor["h"])
351        Resp.build_tracking()
352        ORMH = Resp.response
353        Resp = at.latticetools.OrbitResponseMatrix(ring,"v", ind_bpm, ind_cor["v"])
354        Resp.build_tracking()
355        ORMV = Resp.response
356        
357        #Numerically inverse the ORM
358        kicks["h"], *_ =np.linalg.lstsq(ORMH, dxs)
359        kicks["v"], *_ =np.linalg.lstsq(ORMV, dys)
360        
361        #Aplico com el kick contrari al que està causant el desplaçament observat per corregir-lo
362        kicks["h"] = -kicks["h"]
363        kicks["v"] = -kicks["v"]
364        
365        t_kicks["h"] = t_kicks["h"]+kicks["h"]
366        t_kicks["v"] = t_kicks["v"]+kicks["v"]
367        
368        applyCorrections(ring, ind_cor["h"], kicks["h"], "h")
369        applyCorrections(ring, ind_cor["v"], kicks["v"], "v")
370        
371        orbit = at.find_orbit6(ring, refpts=ind_bpm)[1]
372        dxs = np.array([orbit[i][0] -original_orbit[ind_bpm[i]][0] for i in range(len(orbit))])
373        dys = np.array([orbit[i][2] -original_orbit[ind_bpm[i]][2] for i in range(len(orbit))])
374        
375        difference = rms(dxs)+rms(dys)
376        #print(difference)
377        if difference<threshold:
378            #Restore the ring after changing the kickangles
379            applyCorrections(ring, ind_cor["h"],  -t_kicks["h"], "h")
380            applyCorrections(ring, ind_cor["v"], -t_kicks["v"], "v")
381            return t_kicks, orbit
382        
383    #print("Iteration did not converge")
384    applyCorrections(ring, ind_cor["h"],  -t_kicks["h"], "h")
385    applyCorrections(ring, ind_cor["v"], -t_kicks["v"], "v")
386    return t_kicks, orbit

388def Cor_SVD_cor(ring, ind, threshold, original_orbit, ORMH, ORMV, recalc_step=12):
389    """
390    Corrects orbit using Average Dispersion for frequency feedback.
391    """
392    max_steps = 30
393    local_ORMH = ORMH
394    local_ORMV = ORMV
395    
396    t_kicks = {"h": np.zeros(len(ind["cor"]["h"])), "v": np.zeros(len(ind["cor"]["v"]))}
397    t_d_ring_freq = 0.0

400    optics_data = at.get_optics(ring, refpts=range(len(ring)))
401    orbit = optics_data[2]["closed_orbit"]
402    
403    # Extract Arrays for Average Calc
404    # optics_data[2] is a record array. We need columns 'dispersion' (idx 0) and 'dispersion' derivative?
405    # AT usually returns [D, D', ...] in 'dispersion'.

418
419    for step in range(1, max_steps + 1):
420        
421        # --- A. RECALCULATE ORM IF NEEDED ---
422        if step == recalc_step:
423            print(f"Step {step}: Recalculating ORM...")
424            # ... (Same as before) ...
425            Resp_H = at.latticetools.OrbitResponseMatrix(ring, "h", ind["bpm"], ind["cor"]["h"])

446
447        # --- C. SVD CORRECTION ---
448        kicks_h, *_ = np.linalg.lstsq(local_ORMH, dxs, rcond=None)

452        kicks_v = -kicks_v

489        t_kicks["h"] += kicks2
490        
491        # --- E. UPDATE OPTICS & CHECK ---
492        ##########################
493        # Testing without changing the optics!
494        ##########################
495        #optics_data = at.get_optics(ring, refpts=range(len(ring)))
496        orbit = at.find_orbit6(ring, refpts=range(len(ring)))[1]
497        #all_disp = optics_data[2]['dispersion'] # Update for next loop
498        
499        bpm_orbit = orbit[ind["bpm"]]
500        dxs = np.array([bpm_orbit[i][0] - original_orbit[ind["bpm"][i]][0] for i in range(len(bpm_orbit))])
501        dys = np.array([bpm_orbit[i][2] - original_orbit[ind["bpm"][i]][2] for i in range(len(bpm_orbit))])

503        diff = rms(dxs) + rms(dys)

511    return t_kicks, t_d_ring_freq, orbit

514def compute_single_CFD(ring, CFD, ORMH, ORMV, step, ind, closed_orbit, method):
515    """
516    Computes derivatives using Central Difference (f(x+h) - f(x-h)) / 2h.
517    """
518    print("HOLAAAAAA")
519    # GEOMETRY
520    base_element = ring[CFD]
521    length = base_element.Length if getattr(base_element, 'Length', 0) != 0 else 1.0
522    
523    # Bending
524    angle = getattr(base_element, 'BendingAngle', 0.0)
525    B0 = angle / length
526    
527    # Calculate Ratio (Dipole / Quadrupole) to maintain CFD geometry
528    val_quad = base_element.PolynomB[1]
529    
530    if val_quad != 0:
531        ratio = B0 / val_quad
532    else:
533        ratio = 0.0 
534
535    # --- 2. INTERNAL HELPER FOR SIMULATION ---
536    def _get_state(perturbation_sign):
537        """
538        Runs the simulation for a specific perturbation direction (+1 or -1).
539        """
540        local_ring = copy.deepcopy(ring)
541        
542        # Apply Perturbation (+step or -step)
543        delta = step * perturbation_sign
544        local_ring[CFD].PolynomB[1] += delta
545        local_ring[CFD].PolynomB[0] += delta * ratio
546

149def compute_average_dispersion(ring, ind, all_disp):
150    """ Calculates the average dispersion in a set of ring elements.
151    In the horizontal direction
152    Handles thick elements, thin elements, and zero-focusing correctors/drifts safely.
153    """
154    avDispersion = np.zeros(len(ind))
155    dispersion = all_disp[ind, 0]
156    dispersionp = all_disp[ind, 1]
157    
158    for i, idx in enumerate(ind):
159        el = ring[idx]
160        length = getattr(el, "Length", 0.0)
161        bend = getattr(el, "BendingAngle", 0.0)
162        k = getattr(el, "K", 0.0)
163        
164        # If it's a thin element (length = 0), average dispersion is just point dispersion
165        if length == 0:
166            avDispersion[i] = dispersion[i]
167            continue
168        
169        k_total = k + (bend/length)**2
170        
171        # Prevent 0/0 division (NaNs) for standard correctors/drifts
172        if abs(k_total) < 1e-12:
173            # Analytical limit of integration when k -> 0
174            avDispersion[i] = dispersion[i] + dispersionp[i] * length / 2.0 + bend * length / 6.0
175        else:
176            k_c = complex(k_total)
177            phi = np.sqrt(k_c) * length
178            term1 = dispersion[i] * np.sin(phi) / phi
179            term2 = dispersionp[i] / (k_c * length) * (1 - np.cos(phi))
180            term3 = bend / (length * k_c) * (1 - np.sin(phi) / phi)
181            avDispersion[i] = np.real(term1 + term2 + term3)
182            
183    return avDispersion
