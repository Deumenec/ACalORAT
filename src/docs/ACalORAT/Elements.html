 12def broadcast_vector(v, axis, ndim):
 13    """
 14    Takes a numpy 1D vector and returns the numpy tensor [None, ..., :, ..., None]
 15    with the vector broadcasting in the specified axis.
 16    
 17    Parameters
 18    ----------
 19    v    : 1D vector to broadcast
 20    axis : axis along which v should vary
 21    ndim : number of dimensions of the target tensor
 22    """
 23    shape = [1] * ndim
 24    shape[axis] = -1
 25    return v.reshape(shape)

 84        """

 91        all_optics : tuple

115        self.alpha= np.array([all_optics[2]["alpha"][i][dir_ind] for i in ind], dtype= complex)

119        self.dispersionp = np.array([all_optics[2]["dispersion"][i][2*dir_ind+1] for i in ind], dtype= complex)
120        if hasattr(ring[ind[0]], "BendingAngle"):self.Bend = np.array([ring[i].BendingAngle for i in ind])
121        if hasattr(ring[ind[0]], "Length"):      self.Length = np.array([ring[i].Length for i in ind])
122        if (hasattr(ring[ind[0]], "PolynomB" ) and len(ring[ind[0]].PolynomB) >=2):  self.K = np.array([-sgn*ring[i].PolynomB[1] for i in ind], dtype= complex)
123        if (hasattr(ring[ind[0]], "PolynomB" ) and len(ring[ind[0]].PolynomB) >=3):  self.S = np.array([-sgn*ring[i].PolynomB[2] for i in ind], dtype= complex)
124        if hasattr(ring[ind[0]], "EntranceAngle"): self.EntranceAngle = np.array([ring[i].EntranceAngle for i in ind], dtype= complex)   
125        if hasattr(ring[ind[0]], "ExitAngle"): self.ExitAngle = np.array([ring[i].ExitAngle for i in ind], dtype= complex)

127    def broadcasters(self, axis, ndim):
128        """
129        Prepares variables in Elements for broadcasting with numpy.
130        
131        It loops through all the attributes in the class which don't start 
132        with _ and creates an array with the same name and the suffix "B" with
133        the values broadcasting in the necessary direction.
134        
135        Parameters
136        ----------
137        axis : int
138            The axis where the values are aligned
139        ndim : int
140            The total number of axis for the final calculation.
141        
142        Example: bpm.broadcasters(2, 3) adds among all the other optics: muB = mu[None, None, :]
143        
144        """
145        self._bAxis = axis #Saving the broadcasting axis
146        self._ndim  = ndim
147        variables = [attr for attr in self.__dict__ if not attr.startswith('_')] #Crea broadcasters per totes les variables igual
148        for var in variables:
149            setattr(self, var+"B", broadcast_vector( getattr(self, var) , axis, ndim))

151    def correct_entrance(self):
152        """ Applies edge focusing to correct the entrance optics at the start 
153        of a bending magnet due to the thin quadrupole caused by the non-zero 
154        entrance angle.
155        
156        It corrects the alpha, the gamma, the derivative of dispersion 
157        and in the horizontal direction it adds the extra focusing due to bending.
158        
159        Finally, IF the element is broadcasting in some dimension, it rebuilds
160        the broadcasting.
161        """
162        #To prevent the correction from happening multiple times.
163        if hasattr(self, "_corrected"): return
164        setattr(self, "_corrected", True)
165        
166        #Entrance angle correction to betas
167        if hasattr(self, "EntranceAngle"):
168            self.alpha += self._sgn *self.beta*np.tan(self.EntranceAngle)*(self.Bend/self.Length)
169            #Gamma at new optics
170            self.gamma = (1 + self.alpha**2) / self.beta
171            
172            #Dispersion correction in the horizontal dimension
173            if self._sgn ==-1: 
174                self.dispersionp += np.tan(self.EntranceAngle)*(self.Bend/self.Length)*self.dispersion
175                
176        #strength correction for the horizontal dimension
177        if self._sgn == -1 and hasattr(self, "Bend"):
178            if not hasattr(self, "K"):
179                self.K =0
180            self.K +=  (self.Bend/self.Length)*(self.Bend/self.Length)
181            
182        if hasattr(self, "_bAxis"):
183            self.broadcasters(self._bAxis, self._ndim)
184        return
185        
186    def correct_strength(self):
187        """ Calculates the closed orbit off-momentum and uses it to correct the actual effective
188        strength of the quadrupoles... It can be applied before broadcasting but it is a really mince correction... 0.1%
189        
190        In principle also this momentum deviation is already in the same units as beam rigidity
191        """
192        #This term has an almost neglegible effect.
193        co = at.find_orbit6(self._ring, self._ind)[1] 
194        deviations = np.array([i[4] for i in co]) #The momentum deviation is in the 4rth component
195        self.K = self.K/(1+deviations)
196        
197    def average(self):
198        """Computes the average of the dispersion inside of elements in the 
199        family by applying entrance angle, propagating and integrating inside.
200        """
201        #We start by considering the effect of the entrance fringe field:
202        dispersionp = self.dispersionp
203        if hasattr(self, "EntranceAngle"):
204            dispersionp +=  -self.dispersion*np.tan(self.EntranceAngle)*(self.Bend/self.Length)
205        if hasattr(self, "Bend"):
206            k_total = self.K+(self.Bend/self.Length)**2 #Effective K value
207            phi = np.sqrt(k_total)*self.Length
208            self.avDispersion = self.dispersion *np.sin(phi)/phi + dispersionp/(k_total*self.Length)*(1-np.cos(phi))+self.Bend/(self.Length*k_total)*(1-np.sin(phi)/phi)
209        if (not hasattr(self, "Bend")) and hasattr(self, "K"):
210            k_total = self.K
211            phi = np.sqrt(k_total)*self.Length
212            self.avDispersion = self.dispersion + self.dispersionp*self.Length/2
213        if (not hasattr(self, "Bend")) and not hasattr(self, "K"):
214            self.avDispersion = self.dispersion + self.dispersionp*self.Length/2
215        if hasattr(self, "_bAxis"):
216            setattr(self, "avDispersion"+"B", broadcast_vector( self.avDispersion ,self._bAxis, self._ndim))
